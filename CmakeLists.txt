cmake_minimum_required(VERSION 3.10)
cmake_policy(SET CMP0167 NEW)

set(GAME GraphWarEdu)

project(${GAME}
	VERSION 0.1.0
)

include(GNUInstallDirs)


find_package(gf REQUIRED COMPONENTS graphics)


if (UNIX)
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fsanitize=address")
endif ()


set(GAME_DATADIR "${CMAKE_INSTALL_PREFIX}/share/games/${GAME}" CACHE PATH "Game Data Dir")
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/Client/src/config.h.in ${CMAKE_CURRENT_BINARY_DIR}/Client/config.h @ONLY)

add_executable(${GAME}
	Client/src/main.cpp
	Client/src/local/GraphWarEdu.cpp
	Model/Entity.cpp
	Model/Player.cpp
	Model/Map.cpp
	Model/Scenario.cpp
	Model/Game.cpp
	Client/src/local/Buttons.cpp
	Client/src/local/scenes/GameScene.cpp
	Client/src/local/scenes/MenuScene.cpp
	Client/src/local/scenes/RulesScene.cpp
)


target_include_directories(${GAME}
	PRIVATE
		${CMAKE_CURRENT_SOURCE_DIR}/Client/src
		${CMAKE_CURRENT_SOURCE_DIR}/Client/src/local
		${CMAKE_CURRENT_BINARY_DIR}/Client
		${CMAKE_CURRENT_BINARY_DIR}
)

target_link_libraries(${GAME}
	PRIVATE
		gf::graphics
		gf::net
)

install(
	TARGETS ${GAME}
	RUNTIME DESTINATION "${CMAKE_INSTALL_BINDIR}"
)

#data

install(
	DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/Client/data/"
	DESTINATION "${GAME_DATADIR}"

)
